    <script>
        // IIFE for BMI Calculator
        (function() {
            const container = document.querySelector('.bmi-calculator-embed-container');
            if (!container) {
                console.error("BMI Calculator container not found!");
                return;
            }

            const form = container.querySelector('#bmiCalculatorFormEmbed');
            const weightInput = container.querySelector('#bmiWeightEmbed');
            const heightInput = container.querySelector('#bmiHeightEmbed');
            const resultArea = container.querySelector('#bmiResultEmbed');

            if (!form || !weightInput || !heightInput || !resultArea) {
                console.error("BMI Calculator Embed Error: Could not find necessary elements.");
                return; // Stop script if elements are missing
            }

            form.addEventListener('submit', function(event) {
                event.preventDefault(); // Prevent default form submission

                // Clear previous results and styles
                resultArea.innerHTML = '';
                resultArea.className = 'bmi-result-area'; // Reset classes

                // Get and validate inputs
                const weight = parseFloat(weightInput.value);
                const heightCm = parseFloat(heightInput.value);

                 // Basic validation
                if (isNaN(weight) || weight <= 10 || weight > 500 || isNaN(heightCm) || heightCm <= 50 || heightCm > 250) {
                    resultArea.innerHTML = '<span class="bmi-error">Please enter valid weight (10-500 kg) and height (50-250 cm).</span>';
                    resultArea.classList.add('visible', 'obese'); // Use error styling
                    return;
                }

                // Calculate BMI
                const heightM = heightCm / 100;
                const bmi = weight / (heightM * heightM);
                const bmiRounded = bmi.toFixed(1); // Round to one decimal place

                // Determine BMI category
                let category = '';
                let categoryClass = '';

                if (bmi < 18.5) {
                    category = 'Underweight';
                    categoryClass = 'underweight';
                } else if (bmi >= 18.5 && bmi < 24.9) {
                    category = 'Normal weight';
                    categoryClass = 'normal';
                } else if (bmi >= 25 && bmi < 29.9) {
                    category = 'Overweight';
                    categoryClass = 'overweight';
                } else { // bmi >= 30
                    category = 'Obese';
                    categoryClass = 'obese';
                }

                // Display result
                resultArea.innerHTML = `
                    <span class="bmi-value">${bmiRounded}</span>
                    <span class="bmi-category">Your BMI category is: <strong>${category}</strong></span>
                `;
                resultArea.classList.add('visible', categoryClass);
            });
        })(); // End BMI Calculator IIFE


       // Simple Mobile Menu Toggle Script (Same as before)
       (function() {
           const menuButton = document.getElementById('mobile-menu-toggle');
           const mobileNav = document.getElementById('mobile-nav-menu');

           if (menuButton && mobileNav) {
               menuButton.addEventListener('click', function() {
                   mobileNav.classList.toggle('active');
               });

                mobileNav.querySelectorAll('a').forEach(link => {
                    link.addEventListener('click', () => {
                         mobileNav.classList.remove('active');
                    });
                });

                 document.addEventListener('click', function(event) {
                     if (!mobileNav || !menuButton) return;
                     const isClickInsideNav = mobileNav.contains(event.target);
                     const isClickOnButton = menuButton.contains(event.target);

                     if (!isClickInsideNav && !isClickOnButton && mobileNav.classList.contains('active')) {
                         mobileNav.classList.remove('active');
                     }
                 });
           } else {
                console.warn("Mobile menu button or nav element not found.");
           }
       })();
    </script>
