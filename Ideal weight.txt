    <script>
        // IIFE for Ideal Weight Calculator
        (function() {
             const container = document.querySelector('.ideal-weight-calculator-container');
             if (!container) {
                 console.error("Ideal Weight Calculator container not found!");
                 return;
             }

            // Get DOM elements
            const iwcGenderSelect = container.querySelector('#iwc-gender');
            const iwcUnitSelect = container.querySelector('#iwc-units');
            const iwcHeightCmInput = container.querySelector('#iwc-height-cm');
            const iwcHeightFtInput = container.querySelector('#iwc-height-ft');
            const iwcHeightInInput = container.querySelector('#iwc-height-in');
            const iwcCalculateBtn = container.querySelector('#iwc-calculate-btn');
            const iwcResultsDiv = container.querySelector('#iwc-results');
            const iwcErrorMessageDiv = container.querySelector('#iwc-error-message');
            const iwcResultDevine = container.querySelector('#iwc-result-devine');
            const iwcResultRobinson = container.querySelector('#iwc-result-robinson');
            const iwcResultMiller = container.querySelector('#iwc-result-miller');
            const iwcResultHamwi = container.querySelector('#iwc-result-hamwi');
            const iwcHeightMetricDiv = container.querySelector('#iwc-height-metric');
            const iwcHeightImperialDiv = container.querySelector('#iwc-height-imperial');

            // Check if all essential elements exist
            if (!iwcGenderSelect || !iwcUnitSelect || !iwcHeightCmInput || !iwcHeightFtInput || !iwcHeightInInput || !iwcCalculateBtn || !iwcResultsDiv || !iwcErrorMessageDiv || !iwcResultDevine || !iwcResultRobinson || !iwcResultMiller || !iwcResultHamwi || !iwcHeightMetricDiv || !iwcHeightImperialDiv) {
                console.error("One or more elements required for the Ideal Weight Calculator were not found.");
                 if(iwcCalculateBtn) iwcCalculateBtn.disabled = true; // Disable button if elements missing
                return;
            }

            // Function to toggle input visibility based on unit selection
            function iwcToggleUnits() {
                const selectedUnit = iwcUnitSelect.value;
                if (selectedUnit === 'metric') {
                    iwcHeightMetricDiv.style.display = 'block';
                    iwcHeightImperialDiv.style.display = 'none';
                    iwcHeightFtInput.value = ''; // Clear imperial inputs
                    iwcHeightInInput.value = '';
                } else { // imperial
                    iwcHeightMetricDiv.style.display = 'none';
                    iwcHeightImperialDiv.style.display = 'block';
                    iwcHeightCmInput.value = ''; // Clear metric input
                }
                // Hide results and errors when units change
                iwcResultsDiv.style.display = 'none';
                iwcErrorMessageDiv.style.display = 'none';
            }

            // Function to display errors
            function iwcDisplayError(message) {
                iwcErrorMessageDiv.textContent = message;
                iwcErrorMessageDiv.style.display = 'block';
                iwcResultsDiv.style.display = 'none'; // Hide results on error
            }

            // Function to format results
            function formatWeightResult(kgValue) {
                if (isNaN(kgValue) || kgValue <= 0) {
                    return "-- kg";
                }
                const lbsValue = kgValue * 2.20462;
                // Using innerHTML to allow <strong> tag potentially, but careful with XSS if source changes
                return `${kgValue.toFixed(1)} kg <span style="font-size:0.9em; color:#555;">(${lbsValue.toFixed(1)} lbs)</span>`;
            }

            // Calculation function
            function iwcCalculateIdealWeight() {
                iwcErrorMessageDiv.style.display = 'none'; // Clear previous errors
                iwcErrorMessageDiv.textContent = '';

                const gender = iwcGenderSelect.value;
                const units = iwcUnitSelect.value;
                let heightCm, heightInches;

                if (units === 'metric') {
                    heightCm = parseFloat(iwcHeightCmInput.value);
                    if (isNaN(heightCm) || heightCm <= 50 || heightCm > 250) { // Basic validation
                        iwcDisplayError('Please enter a valid height (50-250 cm).');
                        return;
                    }
                    heightInches = heightCm / 2.54;
                } else { // imperial
                    const heightFt = parseInt(iwcHeightFtInput.value);
                    const heightIn = parseInt(iwcHeightInInput.value || 0);

                    if (isNaN(heightFt) || heightFt < 2 || heightFt > 8 || isNaN(heightIn) || heightIn < 0 || heightIn >= 12) { // Basic validation
                        iwcDisplayError('Please enter valid height (2-8 ft, 0-11 in).');
                        return;
                    }
                    heightInches = (heightFt * 12) + heightIn;
                    heightCm = heightInches * 2.54;
                }

                const minHeightInches = 60; // 5 feet
                 if (heightInches < minHeightInches) {
                      console.warn("Height is below 5 feet, formulas might be less accurate.");
                     // Optionally display a non-blocking warning to the user here
                 }
                const inchesOver5Feet = Math.max(0, heightInches - minHeightInches);

                let ibwDevine, ibwRobinson, ibwMiller, ibwHamwi;

                const devineBase = (gender === 'male') ? 50 : 45.5;
                ibwDevine = devineBase + (2.3 * inchesOver5Feet);

                const robinsonBase = (gender === 'male') ? 52 : 49;
                const robinsonFactor = (gender === 'male') ? 1.9 : 1.7;
                ibwRobinson = robinsonBase + (robinsonFactor * inchesOver5Feet);

                const millerBase = (gender === 'male') ? 56.2 : 53.1;
                const millerFactor = (gender === 'male') ? 1.41 : 1.36;
                ibwMiller = millerBase + (millerFactor * inchesOver5Feet); // Miller also uses inches over 60

                const hamwiBase = (gender === 'male') ? 48.0 : 45.5;
                const hamwiFactor = (gender === 'male') ? 2.7 : 2.2;
                ibwHamwi = hamwiBase + (hamwiFactor * inchesOver5Feet);

                // Ensure results > 0
                ibwDevine = Math.max(1, ibwDevine); // Use 1kg as minimum plausible
                ibwRobinson = Math.max(1, ibwRobinson);
                ibwMiller = Math.max(1, ibwMiller);
                ibwHamwi = Math.max(1, ibwHamwi);

                // Display results using innerHTML for formatted output
                iwcResultDevine.innerHTML = formatWeightResult(ibwDevine);
                iwcResultRobinson.innerHTML = formatWeightResult(ibwRobinson);

                   mobileNav.classList.toggle('active');

                    link.addEventListener('click', () => {
                         mobileNav.classList.remove('active');
                    });
                });

                     if (!mobileNav || !menuButton) return;
                     const isClickInsideNav = mobileNav.contains(event.target);
                     const isClickOnButton = menuButton.contains(event.target);

                     if (!isClickInsideNav && !isClickOnButton && mobileNav.classList.contains('active')) {
                         mobileNav.classList.remove('active');
                     }
                 });
           } else {
                console.warn("Mobile menu button or nav element not found.");
           }
       })();
